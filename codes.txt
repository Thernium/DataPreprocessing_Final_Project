    "#Employee Performance Score\n",
    "\n",
    "# normalize each component to 0-1 range first\n",
    "norm_experience = employee_experience / employee_experience.max()\n",
    "norm_training   = employee_training_hours / employee_training_hours.max()\n",
    "norm_attendance = employee_attendance_rate\n",
    "\n",
    "# education score\n",
    "education_score_map = {\n",
    "    \"High School\": 0.1,\n",
    "    \"Bachelor's\":  0.6,\n",
    "    \"Master's\":    0.8,\n",
    "    \"PhD\":         1.0,\n",
    "}\n",
    "norm_education = np.array([education_score_map[e] for e in employee_education])\n",
    "\n",
    "# weighted formula\n",
    "employee_performance_score = np.round(\n",
    "    (norm_experience  * 0.30 +\n",
    "     norm_attendance  * 0.30 +\n",
    "     norm_training    * 0.20 +\n",
    "     norm_education   * 0.20) * 100,\n",
    "    2\n",
    ")\n",
    "\n",
    "print(f\"Max: %{max(employee_performance_score)}\")\n",
    "print(f\"Min: %{min(employee_performance_score)}\")"
